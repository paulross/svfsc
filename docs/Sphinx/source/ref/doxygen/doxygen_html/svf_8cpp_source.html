282  _throw_diff(fpos, data, iter, 0);

314  // Copy rest of new and break
315  while (len) {

317  ++data;
318  ++fpos;
319  --len;

321  }
322  break;
323  }

387  t_map::iterator base_block_iter) {

389  assert(new_data);
390  assert(new_data_len > 0);
391  assert(base_block_iter != m_svf.end());
392  assert(fpos >= base_block_iter->first);

397 #endif
398  // Diff check against base_block_iter
399  // Do the check to end of new_data_len or end of base_block_iter which ever comes first.
400  // Do not increment m_bytes_total as this is existing new_data.
401  // Diff check against base_block_iter
402  size_t write_index_from_block_start = fpos - base_block_iter->first;
403  // Do the check to end of new_data_len or end of base_block_iter which ever comes first.
404  size_t len_check_or_copy = std::min(new_data_len,
405  base_block_iter->second.size() - write_index_from_block_start);
406  if (m_config.compare_for_diff) {
407  if (std::memcmp(base_block_iter->second.data() + write_index_from_block_start, new_data,
408  len_check_or_copy) != 0) {
409  _throw_diff(fpos, new_data, base_block_iter, write_index_from_block_start);
410  }
411  }
412  new_data += len_check_or_copy;
413  fpos += len_check_or_copy;
414  new_data_len -= len_check_or_copy;
415  t_map::iterator next_block_iter = std::next(base_block_iter);
416  while (new_data_len) {
417  if (next_block_iter == m_svf.end()) {
418  // Termination case, copy remainder
419  while (new_data_len) {
420  base_block_iter->second.push_back(*new_data);
421  ++new_data;
422  ++fpos;
423  --new_data_len;
424  m_bytes_total += 1;
425  }
426  break; // Done. Needed because we are going to do erase(next_block_iter) otherwise.
427  } else {
428  // Copy the new_data up to start of next_block_iter or, we have exhausted the new_data.
429  while (new_data_len && fpos < next_block_iter->first) {
430  base_block_iter->second.push_back(*new_data);
431  ++new_data;
432  ++fpos;
433  --new_data_len;
434  m_bytes_total += 1;
435  }
436  }
437  if (new_data_len == 0 and fpos < next_block_iter->first) {
438  // We have exhausted new data and not reached the next block so we are done
439  break;
440  }
441  // Still data to copy, and we are up against the next block which we will coalesce into the base_block_iter.
442  write_index_from_block_start = 0;
443  // Diff check, but also append to base_block_iter.
444  // Do not increment m_bytes_total as this is existing data.
445  len_check_or_copy = std::min(new_data_len, _file_position_immediatly_after_block(next_block_iter) - fpos);
446  if (len_check_or_copy) {
447  if (m_config.compare_for_diff) {
448  if (std::memcmp(next_block_iter->second.data(), new_data, len_check_or_copy) != 0) {
449  _throw_diff(fpos, new_data, base_block_iter, 0);
450  }
451  }
452  // We could push_back either the new_data or the existing next block data. We choose the former.
453  for (size_t i = 0; i < len_check_or_copy; ++i) {
454  base_block_iter->second.push_back(*new_data);
455  ++new_data;
456  ++fpos;
457  --new_data_len;
458  ++write_index_from_block_start;
459  }
460  }
461  // Here either new_data is exhausted or next_block_iter is.
462  // If new_data is exhausted then copy remaining from next_block_iter to base_block_iter.
463  // Do not increment m_bytes_total as this is existing new_data.
464  if (new_data_len == 0) {
465  while (write_index_from_block_start < next_block_iter->second.size()) {
466  base_block_iter->second.push_back(next_block_iter->second[write_index_from_block_start]);
467  ++write_index_from_block_start;
468  }
469  }
470  // New data is not exhausted so erase next_block_iter as we have copied it and move on to the next block.
471  if (m_config.overwrite_on_exit) {
472  next_block_iter->second.assign(next_block_iter->second.size(), OVERWRITE_CHAR);
473  }
474  next_block_iter = m_svf.erase(next_block_iter);
475  }
476  assert(new_data_len == 0);

478  assert(fpos == fpos_end);
479 #endif

562  t_seek_reads SparseVirtualFile::need(t_fpos fpos, size_t len, size_t greedy_length) const noexcept {
563  SVF_ASSERT(integrity() == ERROR_NONE);
564 #ifdef SVF_THREAD_SAFE
565  std::lock_guard<std::mutex> mutex(m_mutex);
566 #endif
567 
568  if (m_svf.empty()) {
569  return {{fpos, greedy_length > len ? greedy_length : len}};
570  }
571  size_t original_len = len;
572  t_fpos fpos_to = fpos + len;
573  t_seek_reads ret;
574  t_map::const_iterator iter = m_svf.upper_bound(fpos);
575  if (iter == m_svf.begin()) {
576  if (fpos + len <= iter->first) {
577  // ^==|
578  // |+++|
579  ret.push_back({fpos, len});
580  fpos += len;
581  // Mark that we are done.
582  len = 0;
583  }
584  } else {
585  // Otherwise check the previous node with std::prev.
586  auto last_fpos = _file_position_immediatly_after_block(std::prev(iter));
587  if (fpos < last_fpos) {
588  // Example, change:
589  // |==| ^==|
590  // |++++++++++++|
591  // to:
592  // |+++++++++|
593  len -= std::min(len, last_fpos - fpos);
594  fpos = std::min(fpos_to, last_fpos);
595  }
596  }
597  // Now walk through the remaining nodes until we have exhausted the read.
598  while (len) {
599  if (iter == m_svf.end() || fpos + len <= iter->first) {
600  // Either:
601  // |==|
602  // |++++++++++++|
603  // or:
604  // |==|
605  // |+++++|
606  ret.emplace_back(fpos, len);
607  fpos += len;
608  len = 0;
609  break;
610  }
611  // We are in this state:
612  // ^=====|
613  // |++++++++++++++|
614  // or:
615  // ^=====|
616  // |++++++++|
617  if (fpos < iter->first) {
618  assert(len >= iter->first - fpos);
619  auto bytes_added = iter->first - fpos;
620  ret.emplace_back(fpos, bytes_added);
621  len -= bytes_added;
622  fpos += bytes_added;
623  }
624  // We are now in this state:
625  // ^=====|
626  // |+++++++|
627  // or:
628  // ^=====|
629  // |++++|
630  assert(fpos == iter->first);
631  if (fpos + len <= _file_position_immediatly_after_block(iter)) {
632  // ^======|
633  // |++++|
634  fpos += len;
635  len = 0;
636  break;
637  } else {
638  // ^======|
639  // |+++++++++|
640  fpos += iter->second.size();
641  len -= iter->second.size();
642  }
643  ++iter;
644  }
645  assert(fpos == fpos_to);
646  assert(len == 0);
647  if (greedy_length && greedy_length > original_len && !ret.empty()) {
648  ret = _minimise_seek_reads(ret, greedy_length);
649  }
650  return ret;
651  }

672  SparseVirtualFile::_minimise_seek_reads(const t_seek_reads &seek_reads, size_t greedy_length) noexcept {
673  assert (greedy_length != 0);
674 
675  t_seek_reads new_seek_reads;
676  for (const t_seek_read &seek_read: seek_reads) {
677  if (new_seek_reads.empty()) {
678  // Add the first greedy block
679  new_seek_reads.emplace_back(seek_read.first, _amount_to_read(seek_read, greedy_length));
680  } else {
681  // Compare with last new block
682  auto last_iter_of_new = new_seek_reads.end();
683  --last_iter_of_new;
684  if (seek_read.first > last_iter_of_new->first + last_iter_of_new->second) {
685  // Add a new greedy block
686  new_seek_reads.emplace_back(seek_read.first, _amount_to_read(seek_read, greedy_length));
687  } else if (seek_read.first + seek_read.second > last_iter_of_new->first + last_iter_of_new->second) {
688  // Extend last block
689  last_iter_of_new->second +=
690  (seek_read.first + seek_read.second) - (last_iter_of_new->first + last_iter_of_new->second);
691  } // Otherwise do nothing, it is covered by greedy.
692  }
693  }
694  return new_seek_reads;
