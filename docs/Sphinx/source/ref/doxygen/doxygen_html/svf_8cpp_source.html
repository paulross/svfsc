178  assert(len > 0);
179  assert(iter != m_svf.end());

185

187  while (len && fpos < iter->first) {
188  // Copy new data 'X' up to start of iter.
189  // ^===========| |=====|
190  // %XXXX+++++++++++++XX++|

193  ++fpos;
194  --len;

196  }

203  if (std::memcmp(iter->second.data.data(), data, delta) != 0) {
204  _throw_diff(fpos, data, iter, 0);
205  }
206  }
207  for (size_t i = 0; i < delta; ++i) {
208  new_value.data.push_back(*data);
209  ++data;
210  }
211  fpos += delta;
212  len -= delta;
213  index_iter += delta;
214  if (_file_position_immediatly_after_block(iter) > fpos_end) {
215  // Copy rest of iter 'Z'
216  assert(len == 0);
217  // ^=========ZZZ
218  // %+++++++++++++|
219  // So append up to the end of iter and (maybe) go round again.
220  while (index_iter < iter->second.data.size()) {
221  new_value.data.push_back(iter->second.data[index_iter]);
222  ++index_iter;
223  }

225  iter->second.data.assign(iter->second.data.size(), OVERWRITE_CHAR);
226  }

228  break;
229  }

231  if (m_config.overwrite_on_exit) {

233  }
234  iter = m_svf.erase(iter);
235  if (iter == m_svf.end() || iter->first > fpos + len) {
236  // Copy rest of new and break
237  while (len) {
238  new_value.data.push_back(*data);
239  ++data;
240  ++fpos;
241  --len;
242  ++m_bytes_total;
243  }
244  break;
245  }
246  }
247  auto size_before_insert = m_svf.size();
248  m_svf.insert({fpos_start, std::move(new_value)});
249  if (m_svf.size() != 1 + size_before_insert) {

252  os << " Unable to insert new block at " << fpos_start;
253  throw Exceptions::ExceptionSparseVirtualFileWrite(os.str());

309  t_map::iterator base_block_iter) {
310  SVF_ASSERT(integrity() == ERROR_NONE);
311  assert(new_data);
312  assert(new_data_len > 0);
313  assert(base_block_iter != m_svf.end());
314  assert(fpos >= base_block_iter->first);
315  assert(fpos <= _file_position_immediatly_after_block(base_block_iter));
316 
317 #ifdef DEBUG
318  size_t fpos_end = fpos + new_data_len;
319 #endif
320  // Diff check against base_block_iter
321  // Do the check to end of new_data_len or end of base_block_iter which ever comes first.
322  // Do not increment m_bytes_total as this is existing new_data.
323  // Diff check against base_block_iter
324  size_t write_index_from_block_start = fpos - base_block_iter->first;
325  // Do the check to end of new_data_len or end of base_block_iter which ever comes first.
326  size_t len_check_or_copy = std::min(new_data_len,
327  base_block_iter->second.data.size() - write_index_from_block_start);
328  if (m_config.compare_for_diff) {
329  if (std::memcmp(base_block_iter->second.data.data() + write_index_from_block_start, new_data,
330  len_check_or_copy) != 0) {
331  _throw_diff(fpos, new_data, base_block_iter, write_index_from_block_start);
332  }
333  }
334  new_data += len_check_or_copy;
335  fpos += len_check_or_copy;
336  new_data_len -= len_check_or_copy;
337  t_map::iterator next_block_iter = std::next(base_block_iter);
338  while (new_data_len) {
339  if (next_block_iter == m_svf.end()) {
340  // Termination case, copy remainder
341  while (new_data_len) {
342  base_block_iter->second.data.push_back(*new_data);
343  ++new_data;
344  ++fpos;
345  --new_data_len;
346  m_bytes_total += 1;
347  }
348  break; // Done. Needed because we are going to do erase(next_block_iter) otherwise.
349  } else {
350  // Copy the new_data up to start of next_block_iter or, we have exhausted the new_data.
351  while (new_data_len && fpos < next_block_iter->first) {
352  base_block_iter->second.data.push_back(*new_data);
353  ++new_data;
354  ++fpos;
355  --new_data_len;
356  m_bytes_total += 1;
357  }
358  }
359  if (new_data_len == 0 and fpos < next_block_iter->first) {
360  // We have exhausted new data and not reached the next block so we are done
361  break;
362  }
363  // Still data to copy, and we are up against the next block which we will coalesce into the base_block_iter.
364  write_index_from_block_start = 0;
365  // Diff check, but also append to base_block_iter.
366  // Do not increment m_bytes_total as this is existing data.
367  len_check_or_copy = std::min(new_data_len, _file_position_immediatly_after_block(next_block_iter) - fpos);
368  if (len_check_or_copy) {
369  if (m_config.compare_for_diff) {
370  if (std::memcmp(next_block_iter->second.data.data(), new_data, len_check_or_copy) != 0) {
371  _throw_diff(fpos, new_data, base_block_iter, 0);
372  }
373  }
374  // We could push_back either the new_data or the existing next block data. We choose the former.
375  for (size_t i = 0; i < len_check_or_copy; ++i) {
376  base_block_iter->second.data.push_back(*new_data);
377  ++new_data;
378  ++fpos;
379  --new_data_len;
380  ++write_index_from_block_start;
381  }
382  }
383  // Here either new_data is exhausted or next_block_iter is.
384  // If new_data is exhausted then copy remaining from next_block_iter to base_block_iter.
385  // Do not increment m_bytes_total as this is existing new_data.
386  if (new_data_len == 0) {
387  while (write_index_from_block_start < next_block_iter->second.data.size()) {
388  base_block_iter->second.data.push_back(next_block_iter->second.data[write_index_from_block_start]);
389  ++write_index_from_block_start;
390  }
391  }
392  // New data is not exhausted so erase next_block_iter as we have copied it and move on to the next block.
393  if (m_config.overwrite_on_exit) {
394  next_block_iter->second.data.assign(next_block_iter->second.data.size(), OVERWRITE_CHAR);
395  }
396  next_block_iter = m_svf.erase(next_block_iter);
397  }
398  assert(new_data_len == 0);
399 #ifdef DEBUG
400  assert(fpos == fpos_end);
401 #endif
402  SVF_ASSERT(integrity() == ERROR_NONE);
403  }

447  void SparseVirtualFile::write(t_fpos fpos, const char *data, size_t len) {
448  SVF_ASSERT(integrity() == ERROR_NONE);
449 #ifdef SVF_THREAD_SAFE
450  std::lock_guard<std::mutex> mutex(m_mutex);
451 #endif
452  // TODO: throw if !data, len == 0
453  if (m_svf.empty() || fpos > _file_position_immediatly_after_end()) {
454  // Simple insert of new data into empty map or a node beyond the end (common case).
455  _write_new_block(fpos, data, len, m_svf.begin());
456  } else {
457  t_map::iterator iter = m_svf.upper_bound(fpos);
458  if (iter != m_svf.begin()) {
459  --iter;
460  }
461  if (iter->first > fpos) {
462  // Insert new block, possibly coalescing existing blocks.
463  // New comes earlier so either create a new block or copy existing block on to it.
464  if (iter->first <= fpos + len) {
465  // Need to coalesce
466  _write_new_append_old(fpos, data, len, iter);
467  } else {
468  // The new block precedes the old one
469  _write_new_block(fpos, data, len, iter);
470  }
471  } else {
472  // Existing block.first is <= fpos
473  if (fpos > _file_position_immediatly_after_block(iter)) {
474  // No overlap so just write new block
475  _write_new_block(fpos, data, len, m_svf.end());
476  } else {
477  // Append new to existing block, possibly coalescing existing blocks.
478  _write_append_new_to_old(fpos, data, len, iter);
479  }
480  }
481  }
482  // Update internals.
483  m_count_write += 1;
484  m_bytes_write += len;
485  m_time_write = std::chrono::system_clock::now();
486  SVF_ASSERT(integrity() == ERROR_NONE);
487  }

576  t_seek_reads SparseVirtualFile::need(t_fpos fpos, size_t len, size_t greedy_length) const noexcept {
577  SVF_ASSERT(integrity() == ERROR_NONE);
578 #ifdef SVF_THREAD_SAFE
579  std::lock_guard<std::mutex> mutex(m_mutex);
580 #endif
581 
582  if (m_svf.empty()) {
583  return {{fpos, greedy_length > len ? greedy_length : len}};
584  }
585  size_t original_len = len;
586  t_fpos fpos_to = fpos + len;
587  t_seek_reads ret;
588  t_map::const_iterator iter = m_svf.upper_bound(fpos);
589  if (iter == m_svf.begin()) {
590  if (fpos + len <= iter->first) {
591  // ^==|
592  // |+++|
593  ret.push_back({fpos, len});
594  fpos += len;
595  // Mark that we are done.
596  len = 0;
597  }
598  } else {
599  // Otherwise check the previous node with std::prev.
600  auto last_fpos = _file_position_immediatly_after_block(std::prev(iter));
601  if (fpos < last_fpos) {
602  // Example, change:
603  // |==| ^==|
604  // |++++++++++++|
605  // to:
606  // |+++++++++|
607  len -= std::min(len, last_fpos - fpos);
608  fpos = std::min(fpos_to, last_fpos);
609  }
610  }
611  // Now walk through the remaining nodes until we have exhausted the read.
612  while (len) {
613  if (iter == m_svf.end() || fpos + len <= iter->first) {
614  // Either:
615  // |==|
616  // |++++++++++++|
617  // or:
618  // |==|
619  // |+++++|
620  ret.emplace_back(fpos, len);
621  fpos += len;
622  len = 0;
623  break;
624  }
625  // We are in this state:
626  // ^=====|
627  // |++++++++++++++|
628  // or:
629  // ^=====|
630  // |++++++++|
631  if (fpos < iter->first) {
632  assert(len >= iter->first - fpos);
633  auto bytes_added = iter->first - fpos;
634  ret.emplace_back(fpos, bytes_added);
635  len -= bytes_added;
636  fpos += bytes_added;
637  }
638  // We are now in this state:
639  // ^=====|
640  // |+++++++|
641  // or:
642  // ^=====|
643  // |++++|
644  assert(fpos == iter->first);
645  if (fpos + len <= _file_position_immediatly_after_block(iter)) {
646  // ^======|
647  // |++++|
648  fpos += len;
649  len = 0;
650  break;
651  } else {
652  // ^======|
653  // |+++++++++|
654  fpos += iter->second.data.size();
655  len -= iter->second.data.size();
656  }
657  ++iter;
658  }
659  assert(fpos == fpos_to);
660  assert(len == 0);
661  if (greedy_length && greedy_length > original_len && !ret.empty()) {
662  ret = _minimise_seek_reads(ret, greedy_length);
663  }
664  return ret;
665  }

685  t_seek_reads
686  SparseVirtualFile::_minimise_seek_reads(const t_seek_reads &seek_reads, size_t greedy_length) noexcept {
687  assert (greedy_length != 0);
688 
689  t_seek_reads new_seek_reads;
690  for (const t_seek_read &seek_read: seek_reads) {
691  if (new_seek_reads.empty()) {
692  // Add the first greedy block
693  new_seek_reads.emplace_back(seek_read.first, _amount_to_read(seek_read, greedy_length));
694  } else {
695  // Compare with last new block
696  auto last_iter_of_new = new_seek_reads.end();
697  --last_iter_of_new;
698  if (seek_read.first > last_iter_of_new->first + last_iter_of_new->second) {
699  // Add a new greedy block
700  new_seek_reads.emplace_back(seek_read.first, _amount_to_read(seek_read, greedy_length));
701  } else if (seek_read.first + seek_read.second > last_iter_of_new->first + last_iter_of_new->second) {
702  // Extend last block
703  last_iter_of_new->second +=
704  (seek_read.first + seek_read.second) - (last_iter_of_new->first + last_iter_of_new->second);
705  } // Otherwise do nothing, it is covered by greedy.
706  }
707  }
708  return new_seek_reads;

848  SparseVirtualFile::ERROR_CONDITION
849  SparseVirtualFile::integrity() const noexcept {
850  t_fpos prev_fpos;
851  size_t prev_size;
852  t_map::const_iterator iter = m_svf.begin();
853  size_t byte_count = 0;
854 
855  while (iter != m_svf.end()) {
856  t_fpos fpos = iter->first;
857  size_t size = iter->second.data.size();
858  if (iter->second.data.empty()) {
859  return ERROR_EMPTY_BLOCK;
860  }
861  if (iter != m_svf.begin() && fpos == prev_fpos && size == prev_size) {
862  return ERROR_DUPLICATE_BLOCK;
863  }
864  if (iter != m_svf.begin() && prev_fpos + prev_size == iter->first) {
865  return ERROR_ADJACENT_BLOCKS;
866  }
